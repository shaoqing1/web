使用工厂方法创建对象

function createPerson(name, age, gender) {
  //通过该方法，可以创建大批量的对象
  var obj = new Object();
  
  //向对象中添加属性
  obj.name = name
  obj.age = age
  obj.gender = gender
  obj.sayName = function() {
    console.log(this.name)
  }
  return obj;
}
