var obj = new Object();
obj.name = "孙悟空"
obj.age = 18
obj.sayName = function() {
  console.log(obj.name)
}

函数也可以称为对象的属性
  如果一个函数作为一个对象的属性保存
  那么我们称这个函数是这个对像的方法
  调用这个函数就说调用这个对象的方法
  但是这只是名称的区别
  
var obj2 = {
  name: "猪八戒"，
  age: 18,
  sayName: function(){
    console.log(obj2.name)
  }
}

二、枚举对象的属性

var obj = {
  name:"孙悟空",
  age: 18,
  gender: "男",
  address: "花果山"
}

枚举对象中的属性
使用for...in语句
语法：
  for (var 变量 in 对象){}
for in在对象中有多少属性，循环就执行多少次
每次执行时会将属性的名字赋值给变量
